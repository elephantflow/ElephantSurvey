\section{Datasets and Benchmarks}
\label{sec:datasets}

\subsection{Object-Level Datasets}

\textbf{ShapeNet}~\citep{chang2015shapenet}: 51K 3-D CAD models; foundation for
single-image reconstruction training.
\textbf{Objaverse}~\citep{deitke2023objaverse}: 800K+ 3-D objects;
enables LRM-scale training.
\textbf{Objaverse-XL}~\citep{deitke2024objaversexl}: 10M+ 3-D assets.
\textbf{Co3D}~\citep{reizenstein2021co3d}: 19K+ annotated per-category real video
sequences for category-level NVS.
\textbf{MVImgNet}~\citep{yu2023mvimgnet}: 238K multi-view real images across
238 categories.

\subsection{Static Scene Datasets}

\textbf{NeRF-Synthetic}~\citep{mildenhall2020nerf}: 8 synthetic bounded objects.
\textbf{LLFF}~\citep{mildenhall2019llff}: 8 forward-facing real scenes.
\textbf{DTU}~\citep{jensen2014dtu}: 124 controlled MVS scenes.
\textbf{Tanks and Temples}~\citep{knapitsch2017tanks}: large-scale indoor/outdoor.
\textbf{Mip-NeRF~360}~\citep{barron2022mipnerf360}: 9 unbounded 360$^\circ$ scenes.
\textbf{ScanNet}~\citep{dai2017scannet}: 1513 indoor RGB-D sequences.
\textbf{ScanNet++}~\citep{yeshwanth2023scannetpp}: DSLR + iPhone high-resolution indoor.
\textbf{RealEstate10K}~\citep{zhou2018stereoMagnification}: 10K real estate videos
for generalizable NVS training.

\subsection{Dynamic Scene Datasets}

\textbf{D-NeRF}~\citep{pumarola2021dnerf}: 6 monocular synthetic dynamic objects.
\textbf{HyperNeRF}~\citep{park2021hypernerf}: monocular topological changes.
\textbf{NHR}~\citep{wu2020nhr}: multi-view human performance.
\textbf{CMU Panoptic}~\citep{joo2015panoptic}: dense multi-view social activities.
\textbf{Neu3D}~\citep{li2022neural3d}: multi-view dynamic human sequences.
\textbf{iPhone Dataset}~\citep{guo2022neural}: casually-captured phone videos.

\subsection{Autonomous Driving and Large-Scale}

\textbf{Waymo Open}~\citep{sun2020waymo}: multi-camera + LiDAR driving.
\textbf{nuScenes}~\citep{caesar2020nuscenes}: 360$^\circ$ driving + HD maps.
\textbf{KITTI-360}~\citep{liao2022kitti360}: urban driving with panoptic labels.
\textbf{Argoverse~2}~\citep{wilson2023argoverse2}: high-resolution 360$^\circ$ driving.
\textbf{PandaSet}~\citep{xiao2021pandaset}: urban LiDAR+camera dataset.
